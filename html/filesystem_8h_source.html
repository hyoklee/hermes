155  std::size_t hash() const {
156  std::size_t result;
157  std::size_t h1 = std::hash<dev_t>{}(st_dev);
158  std::size_t h2 = std::hash<ino_t>{}(st_ino);
159  std::size_t h3 = std::hash<MPI_File>{}(mpi_fh_);
160  result = h1 ^ h2 ^ h3;
161  return result;
