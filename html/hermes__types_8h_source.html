106  if (policy.find("kRandom") != std::string::npos) {
107  return PlacementPolicy::kRandom;

109  return PlacementPolicy::kRoundRobin;

111  return PlacementPolicy::kMinimizeIoTime;

113  return PlacementPolicy::kNone;
114  }
115  return PlacementPolicy::kNone;
116  }
