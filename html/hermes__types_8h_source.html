104  static PlacementPolicy to_enum(const std::string &policy) {
105  if (policy.find("kRandom") != std::string::npos) {
106  return PlacementPolicy::kRandom;
107  } else if (policy.find("kRoundRobin") != std::string::npos) {
108  return PlacementPolicy::kRoundRobin;

110  return PlacementPolicy::kMinimizeIoTime;
111  } else if (policy.find("kNone") != std::string::npos) {
112  return PlacementPolicy::kNone;
113  }

115  }
