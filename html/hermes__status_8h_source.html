56 #define RETURN_ENUM(ID, NAME, TEXT) NAME = ID,
57 #define RETURN_TEXT(ID, NAME, TEXT) case ID: return TEXT;
58 
60 enum StatusCode {
61  RETURN_CODES(RETURN_ENUM)
62 };
63 
64 namespace api {
65 
67 class Status {
68  public:
70  explicit Status(StatusCode ret_code = HERMES_SUCCESS) : status_(ret_code) {}
71 
73  bool Succeeded() const {
74  return (status_ == HERMES_SUCCESS);
75  }
76 
78  bool Acceptable() const {
79  return (status_ > HERMES_SUCCESS && status_ < HERMES_OK_MAX);
80  }
81 
83  bool Failed() const {
84  return (status_ < HERMES_SUCCESS && status_ > HERMES_ERROR_MAX);
85  }
86 
88  bool operator == (StatusCode code) {
89  return (status_ == code);
90  }
91 
93  void operator = (StatusCode code) {
94  status_ = code;
95  }
96 
98  StatusCode GetStatus() const {return status_;}
99 
101  std::string Msg() const {
102  switch (status_) {
103  RETURN_CODES(RETURN_TEXT)
104  }
105 
106  return "Unknown error";
107  }
108 
109  private:
110  StatusCode status_;
111 };
