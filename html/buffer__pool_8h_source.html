481 template <typename T>
482 std::vector<SwapBlob> PutToSwap(SharedMemoryContext *context, RpcContext *rpc,
483  BucketID id, std::vector<std::vector<T>> &blobs,
484  std::vector<std::string> &names) {
485  size_t num_blobs = blobs.size();
486  std::vector<SwapBlob> result(num_blobs);
487 
488  for (size_t i = 0; i < num_blobs; ++i) {
489  SwapBlob swap_blob =
490  PutToSwap(context, rpc, names[i], id, (const u8 *)blobs[i].data(),
491  blobs[i].size() * sizeof(T));
492  result.push_back(swap_blob);
493  }
494 
495  return result;
496 }
