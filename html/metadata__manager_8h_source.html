47  void InitializeHermes(bool is_mpi) {
48  if (ref == 0) {
49  is_mpi_ = is_mpi;
50  char* async_flush_mode = getenv(kHermesAsyncFlush);
51 
52  if (async_flush_mode && async_flush_mode[0] == '1') {
53  global_flushing_mode = FlushingMode::kAsynchronous;
54  } else {
55  global_flushing_mode = FlushingMode::kSynchronous;
56  }
57 
58  char* hermes_config = getenv(kHermesConf);
59 
60  if (is_mpi_) {
61  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
62  MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
63  // TODO(chogan): Need a better way to distinguish between client and
64  // daemon. https://github.com/HDFGroup/hermes/issues/206
65  if (comm_size > 1) {
66  hermes = hermes::InitHermesClient(hermes_config);
67  } else {
68  is_mpi_ = false;
69  hermes = hermes::InitHermesDaemon(hermes_config);
70  }
71  } else {
72  hermes = hermes::InitHermesDaemon(hermes_config);
73  }
74  }
75  ref++;
76  }
77 
83  void FinalizeHermes() {
84  if (ref == 1) {
85  hermes->FinalizeClient();
86  }
87  ref--;
88  }
