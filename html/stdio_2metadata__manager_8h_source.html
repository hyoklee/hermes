76  void InitializeHermes(bool is_mpi = false) {
77  if (ref == 0) {
78  this->is_mpi = is_mpi;
79  char* async_flush_mode = getenv(kHermesAsyncFlush);
80 
81  if (async_flush_mode && async_flush_mode[0] == '1') {
82  global_flushing_mode = FlushingMode::kAsynchronous;
83  } else {
84  global_flushing_mode = FlushingMode::kSynchronous;
85  }
86 
87  char* hermes_config = getenv(kHermesConf);
88 
89  LOG(INFO) << "InitializeHermes (stdio): " << this->is_mpi << std::endl;
90  if (this->is_mpi) {
91  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
92  MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
93  // TODO(chogan): Need a better way to distinguish between client and
94  // daemon. https://github.com/HDFGroup/hermes/issues/206
95  if (comm_size > 1) {
96  hermes = hermes::InitHermesClient(hermes_config);
97  } else {
98  this->is_mpi = false;
99  hermes = hermes::InitHermesDaemon(hermes_config);
100  }
101  } else {
102  hermes = hermes::InitHermesDaemon(hermes_config);
103  }
104  INTERCEPTOR_LIST->SetupAdapterMode();
105  }
106  ref++;
107  }
112  void FinalizeHermes() {
113  if (ref == 1) {
114  if (this->is_mpi) {
115  MPI_Barrier(MPI_COMM_WORLD);
116  char* stop_daemon = getenv(kStopDaemon);
117  bool shutdown_daemon = true;
118 
119  if (stop_daemon && stop_daemon[0] == '0') {
120  shutdown_daemon = false;
121  }
122 
123  hermes->FinalizeClient(shutdown_daemon);
124  } else {
125  hermes->Finalize(true);
126  }
127  }
128  ref--;
129  }
