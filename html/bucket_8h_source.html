449  const std::vector<std::string> &names,
450  const Context &ctx) {
451  Status result;
452 
453  for (size_t i = 0; i < schemas.size(); ++i) {
454  PlacementSchema &schema = schemas[i];
455  hermes::Blob blob = {};
456  blob.data = (u8 *)blobs[i].data();
457  blob.size = blobs[i].size() * sizeof(T);
458  LOG(INFO) << "Attaching blob '" << names[i] << "' to Bucket '" << name_
459  << "'" << std::endl;
460  result = PlaceBlob(&hermes_->context_, &hermes_->rpc_, schema, blob,

462  if (result.Failed()) {
463  // TODO(chogan): Need to return a std::vector<Status>
464  break;

499 template <typename T>
500 Status Bucket::Put(const std::vector<std::string> &names,
501  const std::vector<std::vector<T>> &blobs,
502  const Context &ctx) {
503  Status ret;
504 
505  for (auto &name : names) {
506  if (IsBlobNameTooLong(name)) {
507  ret = BLOB_NAME_TOO_LONG;
508  LOG(ERROR) << ret.Msg();
509  return ret;
510  }
511  }
512 
513  if (blobs.size() == 0) {
514  ret = INVALID_BLOB;
515  LOG(ERROR) << ret.Msg();
516  return ret;
517  }
518 
519  if (IsValid()) {
520  size_t num_blobs = blobs.size();
521  std::vector<size_t> sizes_in_bytes(num_blobs);
522  for (size_t i = 0; i < num_blobs; ++i) {
523  sizes_in_bytes[i] = blobs[i].size() * sizeof(T);
524  }
525 
526  if (ctx.rr_retry) {
527  int num_devices =
528  GetLocalSystemViewState(&hermes_->context_)->num_devices;
529 
530  for (int i = 0; i < num_devices; ++i) {
531  ret = PutInternal(names, sizes_in_bytes, blobs, ctx);
532 
533  if (ret.Failed()) {
534  RoundRobin rr_state;
535  int current = rr_state.GetCurrentDeviceIndex();
536  rr_state.SetCurrentDeviceIndex((current + 1) % num_devices);
537  } else {
538  break;
539  }
540  }
541  } else {
542  ret = PutInternal(names, sizes_in_bytes, blobs, ctx);
543  }
544  } else {
545  ret = INVALID_BUCKET;
546  LOG(ERROR) << ret.Msg();
547  return ret;
548  }
549 
550  return ret;
551 }
