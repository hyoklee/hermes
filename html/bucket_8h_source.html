262 Status Bucket::PlaceBlobs(std::vector<PlacementSchema> &schemas,
263  const std::vector<std::vector<T>> &blobs,
264  const std::vector<std::string> &names,
265  const Context &ctx) {
266  Status result;
267 
268  for (size_t i = 0; i < schemas.size(); ++i) {
269  PlacementSchema &schema = schemas[i];
270  hermes::Blob blob = {};
271  blob.data = (u8 *)blobs[i].data();
272  blob.size = blobs[i].size() * sizeof(T);
273  LOG(INFO) << "Attaching blob '" << names[i] << "' to Bucket '" << name_
274  << "'" << std::endl;
275  result = PlaceBlob(&hermes_->context_, &hermes_->rpc_, schema, blob,
276  names[i], id_, ctx);
277  if (result.Failed()) {
278  // TODO(chogan): Need to return a std::vector<Status>
279  break;
280  }
281  }
282 
283  return result;
284 }

294 template<typename T>
295 Status Bucket::PutInternal(const std::vector<std::string> &names,
296  const std::vector<size_t> &sizes,
297  const std::vector<std::vector<T>> &blobs,
298  const Context &ctx) {
299  std::vector<PlacementSchema> schemas;
300  HERMES_BEGIN_TIMED_BLOCK("CalculatePlacement");
301  Status result = CalculatePlacement(&hermes_->context_, &hermes_->rpc_, sizes,
302  schemas, ctx);
303  HERMES_END_TIMED_BLOCK();
304 
305  if (result.Succeeded()) {
306  result = PlaceBlobs(schemas, blobs, names, ctx);
307  } else {
308  LOG(ERROR) << result.Msg();
309  }
310 
311  return result;
312 }

314 template<typename T>
315 Status Bucket::Put(const std::vector<std::string> &names,
316  const std::vector<std::vector<T>> &blobs,
317  const Context &ctx) {
318  Status ret;
319 
320  for (auto &name : names) {
321  if (IsBlobNameTooLong(name)) {
322  ret = BLOB_NAME_TOO_LONG;
323  LOG(ERROR) << ret.Msg();
324  return ret;
325  }
326  }
327 
328  if (blobs.size() == 0) {
329  ret = INVALID_BLOB;
330  LOG(ERROR) << ret.Msg();
331  return ret;
332  }
333 
334  if (IsValid()) {
335  size_t num_blobs = blobs.size();
336  std::vector<size_t> sizes_in_bytes(num_blobs);
337  for (size_t i = 0; i < num_blobs; ++i) {
338  sizes_in_bytes[i] = blobs[i].size() * sizeof(T);
339  }
340 
341  if (ctx.rr_retry) {
342  int num_devices =
343  GetLocalSystemViewState(&hermes_->context_)->num_devices;
344 
345  for (int i = 0; i < num_devices; ++i) {
346  ret = PutInternal(names, sizes_in_bytes, blobs, ctx);
347 
348  if (ret.Failed()) {
349  RoundRobinState rr_state;
350  int current = rr_state.GetCurrentDeviceIndex();
351  rr_state.SetCurrentDeviceIndex((current + 1) % num_devices);
352  } else {
353  break;
354  }
355  }
356  } else {
357  ret = PutInternal(names, sizes_in_bytes, blobs, ctx);
358  }
359  } else {
360  ret = INVALID_BUCKET;
361  LOG(ERROR) << ret.Msg();
362  return ret;
363  }
364 
365  return ret;
366 }
