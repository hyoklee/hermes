110  void SetupAdapterMode() {
111  char* adapter_mode_str = getenv(kAdapterMode);
112  if (adapter_mode_str == nullptr) {
113  // default is Persistent mode
114  adapter_mode = AdapterMode::kDefault;
115  } else {
116  if (strcmp(kAdapterDefaultMode, adapter_mode_str) == 0) {
117  adapter_mode = AdapterMode::kDefault;
118  } else if (strcmp(kAdapterBypassMode, adapter_mode_str) == 0) {
119  adapter_mode = AdapterMode::kBypass;
120  } else if (strcmp(kAdapterScratchMode, adapter_mode_str) == 0) {
121  adapter_mode = AdapterMode::kScratch;
122  } else {
123  // TODO(hari): @errorhandling throw error.
124  return;
125  }
126  }
127  char* adapter_paths_str = getenv(kAdapterModeInfo);
128  std::vector<std::string> paths_local;
129  if (adapter_paths_str) {
130  paths_local = StringSplit(adapter_paths_str, kPathDelimiter);
131  }
132  adapter_paths = paths_local;
133  }

135  bool Persists(FILE* fh) { return Persists(GetFilenameFromFP(fh)); }
136  bool Persists(int fd) { return Persists(GetFilenameFromFD(fd)); }

138  bool Persists(std::string path) {
139  if (adapter_mode == AdapterMode::kDefault) {
140  if (adapter_paths.empty()) {
141  return true;
142  } else {
143  for (const auto& pth : adapter_paths) {
144  if (path.find(pth) == 0) {
145  return true;
146  }
147  }
148  return false;
149  }
150  } else if (adapter_mode == AdapterMode::kScratch) {
151  if (adapter_paths.empty()) {
152  return false;
153  } else {
154  for (const auto& pth : adapter_paths) {
155  if (path.find(pth) == 0) {
156  return false;
157  }
158  }
159  return true;
160  }
161  }
162  return true;
163  }
