266 static bool IsBoFunction(const char *func_name) {
267  bool result = false;
268  int i = 0;
269 
270  while (func_name && *func_name != '\0' && i < kBoPrefixLength) {
271  if (func_name[i] != kBoPrefix[i]) {
272  break;
273  }
274  ++i;
275  }
276 
277  if (i == kBoPrefixLength) {
278  result = true;
279  }
280 
281  return result;
282 }

284 template <typename ReturnType, typename... Ts>
285 ReturnType RpcCall(RpcContext *rpc, u32 node_id, const char *func_name,
286  Ts... args) {
287  VLOG(1) << "Calling " << func_name << " on node " << node_id << " from node "
288  << rpc->node_id << std::endl;
289  ClientThalliumState *state = GetClientThalliumState(rpc);
290  bool is_bo_func = IsBoFunction(func_name);
291  std::string server_name = GetServerName(rpc, node_id, is_bo_func);
292 
293  if (is_bo_func) {
294  func_name += kBoPrefixLength;
295  }
296 
297  tl::remote_procedure remote_proc = state->engine->define(func_name);
298  // TODO(chogan): @optimization We can save a little work by storing the
299  // endpoint instead of looking it up on every call
300  tl::endpoint server = state->engine->lookup(server_name);
301 
302  if constexpr (std::is_same<ReturnType, void>::value) {
303  remote_proc.disable_response();
304  remote_proc.on(server)(std::forward<Ts>(args)...);
305  } else {
306  ReturnType result = remote_proc.on(server)(std::forward<Ts>(args)...);
307 
308  return result;
309  }
310 }
