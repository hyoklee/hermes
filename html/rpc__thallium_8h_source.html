235 static bool IsBoFunction(const char *func_name) {
236  bool result = false;
237  int i = 0;
238 
239  while (func_name && *func_name != '\0' && i < kBoPrefixLength) {
240  if (func_name[i] != kBoPrefix[i]) {
241  break;
242  }
243  ++i;
244  }
245 
246  if (i == kBoPrefixLength) {
247  result = true;
248  }
249 
250  return result;
251 }

253 template <typename ReturnType, typename... Ts>
254 ReturnType RpcCall(RpcContext *rpc, u32 node_id, const char *func_name,
255  Ts... args) {
256  VLOG(1) << "Calling " << func_name << " on node " << node_id << " from node "
257  << rpc->node_id << std::endl;
258  ClientThalliumState *state = GetClientThalliumState(rpc);
259  bool is_bo_func = IsBoFunction(func_name);
260  std::string server_name = GetServerName(rpc, node_id, is_bo_func);
261 
262  if (is_bo_func) {
263  func_name += kBoPrefixLength;
264  }
265 
266  tl::remote_procedure remote_proc = state->engine->define(func_name);
267  // TODO(chogan): @optimization We can save a little work by storing the
268  // endpoint instead of looking it up on every call
269  tl::endpoint server = state->engine->lookup(server_name);
270 
271  if constexpr (std::is_same<ReturnType, void>::value) {
272  remote_proc.disable_response();
273  remote_proc.on(server)(std::forward<Ts>(args)...);
274  } else {
275  ReturnType result = remote_proc.on(server)(std::forward<Ts>(args)...);
276 
277  return result;
278  }
279 }
